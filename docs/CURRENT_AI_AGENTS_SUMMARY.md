const agentManager = new AgentManager({

    'study-assistant', 
    'insight-generation',
    'learning',
    'discussion',
    'quiz',
    'teacher',
    'auto-learning'

  webSocketPort: 3001,
  logLevel: 'info',
  maxConcurrentTasks: 10,

    responseTime: 5000, // 5 seconds

  }
});
